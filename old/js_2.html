                    function fn_sissar(s,n)
                    {
                        var rt = 0;

                        if(n > 8000){
                            n = 8000;
                        }

                        let s_ascii_arr = s_ascii_plus_arr = s_string_arr= [];

                        // s를 분리하여 배열로 변경해야 겠다.
                        let s_arr = s.slice(); // 문자열 분리
                        for(let i=0; i < s_arr.length; i++){

                            s_ascii_arr[i] = s.charCodeAt(i);

                            // 대문자라면
                            if(65 <= s_ascii_arr[i] && s_ascii_arr[i] <= 90){
                                 rt = fn_upper(s_ascii_arr[i],n);
                            } else if(97 <= s_ascii_arr[i] && s_ascii_arr[i] <= 122){    // 소문자라면
                                rt = fn_lower(s_ascii_arr[i],n);
                            } else if(32 == s_ascii_arr[i] ){
                                rt = 32;
                            }

                            s_string_arr[i] = String.fromCharCode(rt);  // 아스키코드를 문자열로만 바꾸면 됨.
                        }

                        //배열을 문자열로 만들려면
                        const arr_string = s_string_arr.join('');

                        //document.write(arr_string);

                        return arr_string;
                    }

                    function fn_upper(s,n)
                    {
                        var damunja = 0;
                        var sn = s+n;

                        if(sn > 90){
                            var namugi = (s+n)%90;
                            damunja = 65+namugi;
                        }else{
                            damunja = sn;
                        }
                        return damunja;
                    }

                    function fn_lower(s,n)
                    {
                        var damunja = 0;
                        var sn = s+n;

                        if(sn > 122){
                            var namugi = (s+n)%122;
                            damunja = 97+namugi;
                        }else{
                            damunja = sn;
                        }
                        return damunja;
                    }
